⚡ ВАЖНО про ICS-43434

Микрофон выдаёт 32-битные слова, но полезные данные — 24 бита, выровненные по левому краю. На ESP32 нужно читать 32-бит и сдвигать вправо на 8: int16_t sample = (int32_t)(raw >> 8) >> 8; Это даст корректные знаковые 16-бит сэмплы.

Схема питания

Цепочка питания:
USB-C → TP4056 (4.2V max) → Li-Po 2000mAh → IP5306 (5V boost + 3.3V LDO) → ESP32 + MIC

IP5306 — лучший выбор вместо голого AMS1117: встроенный boost-конвертер + защита от переразряда + индикатор уровня. Стоит ~$0.5 оптом.

Формат пакета

/* Структура пакета VoiceMind Transport Protocol v1 */

┌──────────┬───────────┬───────────┬──────────┬─────────────────────────────┬───────┐
│ START    │ SEQ[2]    │ TYPE[1]   │ LEN[2]   │ PAYLOAD[LEN]                │ CRC8  │
│ 0xAA     │ big-end   │ uint8     │ big-end  │ max 1024 bytes              │ uint8 │
│ 1 byte   │ 2 bytes   │ 1 byte    │ 2 bytes  │ variable                    │ 1byte │
└──────────┴───────────┴───────────┴──────────┴─────────────────────────────┴───────┘

/* TYPE коды: */
#define PKT_AUDIO    0x01  // PCM аудио данные
#define PKT_CTRL     0x02  // Управление (start/stop)
#define PKT_ACK      0x03  // Подтверждение
#define PKT_NACK     0x04  // Запрос повтора
#define PKT_STATUS   0x05  // Статус (батарея, режим)
#define PKT_HEARTBEAT0x06  // Keepalive каждые 1с

CRC-8 (Maxim/Dallas алгоритм)

Полином CRC-8/MAXIM
P(x) = x⁸ + x⁵ + x⁴ + 1 → 0x31 (отражённый: 0x8C)
Инициализация: CRC₀ = 0x00
Для каждого байта: CRC = CRC XOR byte, затем 8 итераций сдвига

#define SAMPLE_RATE   16000
#define DMA_BUF_LEN   512     // сэмплов на DMA буфер
#define DMA_BUF_CNT   8       // количество DMA буферов
#define PKT_PAYLOAD   1024   // байт payload за пакет (512 сэмплов × 2)
#define PKT_HEADER    7      // START+SEQ+TYPE+LEN = 7 байт

BluetoothSerial SerialBT;
volatile bool isRecording = false;
uint16_t seqNum = 0;
int32_t rawBuffer[DMA_BUF_LEN];  // 32-бит сырой I2S
int16_t pcmBuffer[DMA_BUF_LEN];  // 16-бит PCM для передачи
uint8_t pktBuffer[PKT_HEADER + PKT_PAYLOAD + 1];

// ─── CRC-8 (Dallas/Maxim) ─────────────────────────────────────
uint8_t crc8(uint8_t* data, size_t len) {
    uint8_t crc = 0x00;
    for (size_t i = 0; i < len; i++) {
        crc ^= data[i];
        for (int j = 0; j < 8; j++) {
            if (crc & 0x80) crc = (crc << 1) ^ 0x31;
            else crc <<= 1;
        }
    }
    return crc;
}

// ─── ИНИЦИАЛИЗАЦИЯ I2S ────────────────────────────────────────
void setupI2S() {
    i2s_config_t cfg = {
        .mode            = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_RX),
        .sample_rate     = SAMPLE_RATE,
        .bits_per_sample = I2S_BITS_PER_SAMPLE_32BIT,  // ICS-43434: 32bit слова
        .channel_format  = I2S_CHANNEL_FMT_ONLY_LEFT,
        .communication_format = I2S_COMM_FORMAT_STAND_I2S,
        .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,
        .dma_buf_count   = DMA_BUF_CNT,
        .dma_buf_len     = DMA_BUF_LEN,
        .use_apll        = false,
        .tx_desc_auto_clear = false,
        .fixed_mclk      = 0
    };
    i2s_pin_config_t pins = {
        .bck_io_num   = I2S_SCK,
        .ws_io_num    = I2S_WS,
        .data_out_num = I2S_PIN_NO_CHANGE,
        .data_in_num  = I2S_SD
    };
    i2s_driver_install(I2S_NUM_0, &cfg, 0, NULL);
    i2s_set_pin(I2S_NUM_0, &pins);
    i2s_set_clk(I2S_NUM_0, SAMPLE_RATE, I2S_BITS_PER_SAMPLE_32BIT, I2S_CHANNEL_MONO);
}

// ─── КОНВЕРТАЦИЯ 32→16 BIT ────────────────────────────────────
// ICS-43434: данные в битах 31-8 (MSB выровнены, 24-bit), биты 7-0 = 0
// Сдвиг: (int32_t)raw >> 8 даёт 24-бит со знаком, ещё >> 8 = 16-бит
void convertTo16bit(int32_t* src, int16_t* dst, size_t n) {
    for (size_t i = 0; i < n; i++) {
        dst[i] = (int16_t)((int32_t)src[i] >> 16);  // правый сдвиг 16 бит
    }
}

// ─── СБОРКА И ОТПРАВКА ПАКЕТА ─────────────────────────────────
void sendAudioPacket(int16_t* pcm, size_t samples) {
    uint16_t payloadLen = samples * sizeof(int16_t);
    pktBuffer[0] = 0xAA;                          // START
    pktBuffer[1] = (seqNum >> 8) & 0xFF;           // SEQ HI
    pktBuffer[2] = seqNum & 0xFF;                  // SEQ LO
    pktBuffer[3] = 0x01;                           // TYPE: AUDIO
    pktBuffer[4] = (payloadLen >> 8) & 0xFF;       // LEN HI
    pktBuffer[5] = payloadLen & 0xFF;              // LEN LO
    memcpy(&pktBuffer[6], pcm, payloadLen);          // PAYLOAD
    pktBuffer[6 + payloadLen] = crc8(pktBuffer, 6 + payloadLen); // CRC
    SerialBT.write(pktBuffer, PKT_HEADER + payloadLen + 1);
    seqNum++;
}

// ─── ПРЕРЫВАНИЕ КНОПКИ ───────────────────────────────────────
void IRAM_ATTR onButtonPress() {
    isRecording = !isRecording;
    digitalWrite(LED_PIN, isRecording ? HIGH : LOW);
}

void setup() {
    Serial.begin(115200);
    pinMode(LED_PIN, OUTPUT);
    pinMode(BTN_PIN, INPUT_PULLUP);
    attachInterrupt(digitalPinToInterrupt(BTN_PIN), onButtonPress, FALLING);
    
    SerialBT.begin("VoiceMind-01");  // Bluetooth имя
    setupI2S();
    Serial.println("VoiceMind ready. Press button to record.");
}

void loop() {
    if (!isRecording || !SerialBT.hasClient()) {
        delay(10);
        return;
    }
    
    size_t bytesRead = 0;
    // Читаем один DMA буфер
    i2s_read(I2S_NUM_0, rawBuffer, DMA_BUF_LEN * sizeof(int32_t), &bytesRead, portMAX_DELAY);
    
    size_t samplesRead = bytesRead / sizeof(int32_t);
    convertTo16bit(rawBuffer, pcmBuffer, samplesRead);
    sendAudioPacket(pcmBuffer, samplesRead);
}

ESP32 / C++// Статусные сообщения устройству ──────────────────────────────
void sendStatusPacket() {
    uint8_t batt = readBatteryPercent();
    uint8_t payload[3] = {
        batt,           // уровень батареи 0-100
        isRecording,    // статус записи
        0x01            // версия прошивки
    };
    uint8_t pkt[7 + 3 + 1];
    pkt[0] = 0xAA; pkt[1] = 0; pkt[2] = seqNum;
    pkt[3] = 0x05;  // PKT_STATUS
    pkt[4] = 0; pkt[5] = 3;
    memcpy(&pkt[6], payload, 3);
    pkt[9] = crc8(pkt, 9);
    SerialBT.write(pkt, sizeof(pkt));
}

// Уровень батареи через делитель напряжения ─────────────────
// GPIO34 — ADC1 CH6 (12 бит, 0-4095 = 0-3.3V)
// Делитель: 100кОм + 100кОм → V_adc = V_bat / 2
// V_bat полная = 4.2V, пустая = 3.0V
uint8_t readBatteryPercent() {
    uint32_t adcSum = 0;
    for (int i = 0; i < 8; i++) adcSum += analogRead(BATT_ADC);
    float v_adc = (adcSum / 8.0f) * (3.3f / 4095.0f);
    float v_bat = v_adc * 2.0f;
    float pct = (v_bat - 3.0f) / (4.2f - 3.0f) * 100.0f;
    return (uint8_t)constrain(pct, 0, 100);
}

Android — Bluetooth Receiver

Kotlinclass BluetoothManager @Inject constructor(
    private val context: Context
) {
    private val adapter = BluetoothAdapter.getDefaultAdapter()
    private var socket: BluetoothSocket? = null
    private var inputStream: InputStream? = null
    
    // SPP UUID стандартный (не менять)
    private val SPP_UUID = UUID.fromString("00001101-0000-1000-8000-00805F9B34FB")
    
    suspend fun connect(deviceAddress: String): Result<Unit> = withContext(Dispatchers.IO) {
        try {
            val device = adapter.getRemoteDevice(deviceAddress)
            socket = device.createRfcommSocketToServiceRecord(SPP_UUID)
            adapter.cancelDiscovery()
            socket?.connect()
            inputStream = socket?.inputStream
            Result.success(Unit)
        } catch (e: IOException) {
            Result.failure(e)
        }
    }
    
    // Flow который непрерывно эмитирует распарсенные пакеты
    fun audioPacketFlow(): Flow<AudioPacket> = flow {
        val parser = PacketParser()
        val readBuf = ByteArray(2048)
        
        while (true) {
            val bytesRead = inputStream?.read(readBuf) ?: break
            val packets = parser.feed(readBuf, bytesRead)
            packets.forEach { emit(it) }
        }
    }.flowOn(Dispatchers.IO)
}

// ─── ПАРСЕР ПАКЕТОВ (state machine) ──────────────────────────
class PacketParser {
    private enum class State { WAIT_START, READ_SEQ, READ_TYPE, READ_LEN, READ_PAYLOAD, READ_CRC }
    private var state = State.WAIT_START
    private var seq = 0; private var type = 0
    private var expectedLen = 0; private var bytesRead = 0
    private val headerBuf = ByteArray(6)
    private val payloadBuf = ByteArray(1024)
    
    fun feed(data: ByteArray, len: Int): List<AudioPacket> {
        val result = mutableListOf<AudioPacket>()
        for (i in 0 until len) {
            val b = data[i].toInt() and 0xFF
            when (state) {
                State.WAIT_START -> if (b == 0xAA) state = State.READ_SEQ
                State.READ_SEQ -> {
                    headerBuf[bytesRead++] = b.toByte()
                    if (bytesRead == 2) {
                        seq = ((headerBuf[0].toInt() and 0xFF) shl 8) or (headerBuf[1].toInt() and 0xFF)
                        bytesRead = 0; state = State.READ_TYPE
                    }
                }
                State.READ_TYPE -> { type = b; state = State.READ_LEN }
                State.READ_LEN -> {
                    headerBuf[bytesRead++] = b.toByte()
                    if (bytesRead == 2) {
                        expectedLen = ((headerBuf[0].toInt() and 0xFF) shl 8) or (headerBuf[1].toInt() and 0xFF)
                        bytesRead = 0; state = State.READ_PAYLOAD
                    }
                }
                State.READ_PAYLOAD -> {
                    payloadBuf[bytesRead++] = b.toByte()
                    if (bytesRead == expectedLen) { bytesRead = 0; state = State.READ_CRC }
                }
                State.READ_CRC -> {
                    // Проверяем CRC
                    val computed = crc8(payloadBuf, expectedLen)
                    if (computed == b.toByte() && type == 0x01) {
                        result.add(AudioPacket(seq, payloadBuf.copyOf(expectedLen)))
                    }
                    state = State.WAIT_START; bytesRead = 0
                }
            }
        }
        return result
    }
    
    private fun crc8(data: ByteArray, len: Int): Byte {
        var crc = 0
        for (i in 0 until len) {
            crc = crc xor (data[i].toInt() and 0xFF)
            repeat(8) { crc = if (crc and 0x80 != 0) (crc shl 1) xor 0x31 else crc shl 1 }
        }
        return crc.toByte()
    }
}

data class AudioPacket(val seq: Int, val pcmData: ByteArray)

Скачай vosk-model-small-ru-0.22 (~50MB) с alphacephei.com/vosk/models. Положи в assets/vosk-model-small-ru-0.22/. При первом запуске приложение копирует её во внутреннее хранилище (необходимо для нативной библиотеки).

Kotlinclass VoskRecognizer @Inject constructor(
    @ApplicationContext private val context: Context,
    private val scope: CoroutineScope
) {
    private val MODEL_NAME = "vosk-model-small-ru-0.22"
    private var model: Model? = null
    private var recognizer: KaldiRecognizer? = null
    private val SAMPLE_RATE = 16000.0f
    
    suspend fun initialize(): Boolean = withContext(Dispatchers.IO) {
        try {
            val modelPath = extractModelIfNeeded()
            model = Model(modelPath)
            recognizer = KaldiRecognizer(model!!, SAMPLE_RATE).apply {
                setWords(true)          // включить word timestamps
                setPartialWords(true)   // частичные результаты
                setMaxAlternatives(3)    // топ-3 варианта
            }
            true
        } catch (e: Exception) { false }
    }
    
    /**
     * Обрабатывает PCM chunk (ByteArray, little-endian signed 16-bit)
     * Возвращает null если нет результата, иначе RecognitionResult
     */
    fun processChunk(pcmBytes: ByteArray): RecognitionResult? {
        val rec = recognizer ?: return null
        return if (rec.acceptWaveForm(pcmBytes, pcmBytes.size)) {
            parseFullResult(rec.result)
        } else {
            parsePartialResult(rec.partialResult)
        }
    }
    
    private fun parseFullResult(json: String): RecognitionResult {
        val obj = JSONObject(json)
        val text = obj.optString("text", "")
        val words = mutableListOf<WordTiming>()
        obj.optJSONArray("result")?.let { arr ->
            for (i in 0 until arr.length()) {
                val w = arr.getJSONObject(i)
                words.add(WordTiming(
                    word  = w.getString("word"),
                    start = w.getDouble("start").toFloat(),
                    end   = w.getDouble("end").toFloat(),
                    conf  = w.getDouble("conf").toFloat()
                ))
            }
        }
        return RecognitionResult(isFinal = true, text = text, words = words)
    }
    
    private fun parsePartialResult(json: String): RecognitionResult? {
        val partial = JSONObject(json).optString("partial", "")
        return if (partial.isNotEmpty()) RecognitionResult(isFinal = false, text = partial) else null
    }
    
    // Копирует модель из assets во внутреннее хранилище (если ещё нет)
    private suspend fun extractModelIfNeeded(): String = withContext(Dispatchers.IO) {
        val dest = File(context.filesDir, MODEL_NAME)
        if (!dest.exists()) {
            dest.mkdirs()
            context.assets.list(MODEL_NAME)?.forEach { file ->
                context.assets.open("$MODEL_NAME/$file").use { input ->
                    File(dest, file).outputStream().use { input.copyTo(it) }
                }
            }
        }
        dest.absolutePath
    }
}

data class RecognitionResult(
    val isFinal: Boolean,
    val text: String,
    val words: List<WordTiming> = emptyList()
)

Алгоритм VAD (Voice Activity Detection)

Energy-based VAD
E_frame = √( (1/N) × Σᵢ xᵢ² ) — RMS энергия фрейма
E_smooth = α × E_frame + (1-α) × E_smooth_prev — EMA сглаживание

Параметры: α = 0.3, порог = 500 (для 16-bit PCM), пауза = 30 фреймов (~1с)

Фрейм = 512 сэмплов @ 16kHz = 32 мс. Буфер предречи = 10 фреймов (320мс) 
Kotlinclass VADProcessor(
    private val energyThreshold: Float = 500f,
    private val silenceFramesMax: Int = 30,    // ~960ms до конца сегмента
    private val preSpeechPad: Int = 10          // буфер перед речью
) {
    private var state = VADState.SILENCE
    private var silenceCount = 0
    private var smoothedEnergy = 0f
    private val EMA_ALPHA = 0.3f
    
    // Кольцевой буфер для предречевого паддинга
    private val preBuffer = ArrayDeque<ShortArray>(preSpeechPad)
    
    fun processFrame(samples: ShortArray): VADEvent {
        // 1. Вычислить RMS энергию
        val rms = computeRMS(samples)
        
        // 2. EMA сглаживание
        smoothedEnergy = EMA_ALPHA * rms + (1f - EMA_ALPHA) * smoothedEnergy
        
        // 3. State machine
        return when (state) {
            VADState.SILENCE -> {
                preBuffer.addLast(samples.clone())
                if (preBuffer.size > preSpeechPad) preBuffer.removeFirst()
                
                if (smoothedEnergy > energyThreshold) {
                    state = VADState.SPEECH
                    silenceCount = 0
                    val prePad = preBuffer.toList()
                    preBuffer.clear()
                    VADEvent.SpeechStart(prePad)
                } else VADEvent.Silence
            }
            VADState.SPEECH -> {
                if (smoothedEnergy < energyThreshold) {
                    silenceCount++
                    if (silenceCount >= silenceFramesMax) {
                        state = VADState.SILENCE
                        VADEvent.SpeechEnd
                    } else VADEvent.TrailingSilence(samples)
                } else {
                    silenceCount = 0
                    VADEvent.SpeechFrame(samples)
                }
            }
        }
    }
    
    private fun computeRMS(samples: ShortArray): Float {
        var sumSq = 0.0
        samples.forEach { sumSq += (it.toDouble() * it.toDouble()) }
        return sqrt(sumSq / samples.size).toFloat()
    }
}

sealed class VADEvent {
    data class SpeechStart(val prePadFrames: List<ShortArray>): VADEvent()
    data class SpeechFrame(val samples: ShortArray): VADEvent()
    data class TrailingSilence(val samples: ShortArray): VADEvent()
    object SpeechEnd : VADEvent()
    object Silence : VADEvent()
}

Kotlin// ─── TF-IDF KEYWORD EXTRACTOR ────────────────────────────────
class KeywordExtractor {
    private val STOPWORDS = setOf(
        "и", "в", "на", "с", "по", "из", "что", "как", "это", "он",
        "она", "мы", "вы", "они", "я", "но", "а", "же", "бы", "не",
        "то", "да", "нет", "ну", "вот", "уже", "или", "при", "для",
        "под", "над", "об", "все", "так", "тут", "там"
    )
    
    /**
     * TF-IDF для одного документа (упрощённый: IDF = 1 для всех слов в одном doc)
     * TF(t, d) = count(t в d) / count(всех слов в d)
     * Для многодокументного корпуса используем сохранённые IDF из БД
     */
    fun extractKeywords(text: String, topN: Int = 10, idfMap: Map<String, Float> = emptyMap()): List<Keyword> {
        val tokens = text.lowercase()
            .split(Regex("""[\s,.:;!?()"'\n—–]+"""))
            .filter { it.length > 2 && it !in STOPWORDS && it.all { c -> c.isLetter() } }
        
        val totalWords = tokens.size.toFloat()
        if (totalWords == 0f) return emptyList()
        
        val tf = tokens.groupingBy { it }.eachCount()
            .mapValues { (_, count) -> count.toFloat() / totalWords }
        
        return tf.entries
            .map { (word, tfScore) ->
                val idf = idfMap[word] ?: 1.0f
                Keyword(word, tfScore * idf, tf[word] ?: 0)
            }
            .sortedByDescending { it.score }
            .take(topN)
    }
}

// ─── TEXTRANK SUMMARIZER ─────────────────────────────────────
class TextRankSummarizer(
    private val damping: Double = 0.85,
    private val iterations: Int = 20
) {
    /**
     * TextRank для извлечения ключевых предложений.
     * 
     * Алгоритм:
     * 1. Разбиваем на предложения
     * 2. Считаем матрицу схожести S[i][j] = |intersection(sᵢ, sⱼ)| / (log|sᵢ| + log|sⱼ|)
     * 3. Нормализуем строки матрицы (→ stochastic matrix)
     * 4. Итерируем: rank = (1-d)/N + d × Σⱼ(S[j][i] × rank[j])
     * 5. Возвращаем топ-K предложений по рангу, в оригинальном порядке
     */
    fun summarize(text: String, topK: Int = 3): String {
        val sentences = text.split(Regex("""[.!?]+\s*"""))
            .map { it.trim() }
            .filter { it.split(" ").size > 4 }
        
        if (sentences.size <= topK) return sentences.joinToString(". ")
        
        val N = sentences.size
        val wordSets = sentences.map { it.lowercase().split(" ").toSet() }
        
        // Матрица схожести
        val sim = Array(N) { i ->
            DoubleArray(N) { j ->
                if (i == j) 0.0 else {
                    val intersection = wordSets[i].intersect(wordSets[j]).size.toDouble()
                    val denom = ln(wordSets[i].size.toDouble()) + ln(wordSets[j].size.toDouble())
                    if (denom > 0) intersection / denom else 0.0
                }
            }
        }
        
        // Нормализация строк
        for (i in 0 until N) {
            val rowSum = sim[i].sum()
            if (rowSum > 0) sim[i].indices.forEach { j -> sim[i][j] /= rowSum }
        }
        
        // PageRank итерации
        var ranks = DoubleArray(N) { 1.0 / N }
        repeat(iterations) {
            val newRanks = DoubleArray(N) { i ->
                (1.0 - damping) / N + damping * (0 until N).sumOf { j -> sim[j][i] * ranks[j] }
            }
            ranks = newRanks
        }
        
        // Выбираем топ-K, сохраняем оригинальный порядок
        val topIndices = ranks.indices.sortedByDescending { ranks[it] }.take(topK).toSet()
        return sentences.indices
            .filter { it in topIndices }
            .joinToString(". ") { sentences[it] }
    }
}

// ─── NLP PIPELINE (главный класс) ─────────────────────────────
class NLPPipeline @Inject constructor(
    private val keywordExtractor: KeywordExtractor,
    private val summarizer: TextRankSummarizer
) {
    private val TASK_PATTERNS = listOf(
        Regex("""(?:нужно|необходимо|надо|следует|должн[ыо])\s+(.{5,50})""", RegexOption.IGNORE_CASE),
        Regex("""(?:сделать|подготовить|проверить|отправить|встретиться)\s+(.{5,50})""", RegexOption.IGNORE_CASE),
        Regex("""(?:договорились|решили|согласовали)\s+(?:что\s+)?(.{5,50})""", RegexOption.IGNORE_CASE)
    )
    
    private val DATE_PATTERN = Regex(
        """(\d{1,2})[.\s/](\d{1,2})[.\s/]?(\d{2,4})?|""" +
        """((?:январ|феврал|март|апрел|май|июн|июл|август|сентябр|октябр|ноябр|декабр)\w*)""",
        RegexOption.IGNORE_CASE
    )
    
    private val MEETING_KW  = setOf("встреча", "переговоры", "договорились", "участник", "согласовали", "повестка")
    private val LECTURE_KW  = setOf("лекция", "тема", "студент", "рассмотрим", "изучаем", "конспект", "семинар")
    private val INTERVIEW_KW = setOf("интервью", "кандидат", "опыт", "расскажите", "вакансия", "должность")
    
    fun process(rawText: String): StructuredNote {
        val keywords = keywordExtractor.extractKeywords(rawText, topN = 15)
        val summary   = summarizer.summarize(rawText, topK = 3)
        val tasks      = extractTasks(rawText)
        val dates      = extractDates(rawText)
        val type       = classifyType(rawText)
        val title      = generateTitle(keywords, type)
        
        return StructuredNote(title, type, summary, keywords, tasks, dates, rawText)
    }
    
    private fun extractTasks(text: String): List<String> =
        TASK_PATTERNS.flatMap { pattern ->
            pattern.findAll(text).map { it.groupValues[1].trim() }.toList()
        }.distinct().take(10)
    
    private fun extractDates(text: String): List<String> =
        DATE_PATTERN.findAll(text).map { it.value.trim() }.distinct().take(10).toList()
    
    private fun classifyType(text: String): ContentType {
        val lower = text.lowercase()
        val scores = mapOf(
            ContentType.MEETING   to MEETING_KW.count { it in lower },
            ContentType.LECTURE   to LECTURE_KW.count { it in lower },
            ContentType.INTERVIEW to INTERVIEW_KW.count { it in lower }
        )
        val best = scores.maxByOrNull { it.value }
        return if ((best?.value ?: 0) > 0) best!!.key else ContentType.OTHER
    }
    
    private fun generateTitle(keywords: List<Keyword>, type: ContentType): String {
        val typeName = when(type) {
            ContentType.MEETING   -> "Встреча"
            ContentType.LECTURE   -> "Лекция"
            ContentType.INTERVIEW -> "Интервью"
            ContentType.OTHER     -> "Запись"
        }
        val topWords = keywords.take(2).joinToString(", ") { it.word.replaceFirstChar(Char::uppercase) }
        return if (topWords.isEmpty()) "$typeName от ${Date()}" else "$typeName: $topWords"
    }
}

data class StructuredNote(

    val keywords: List<Keyword>, val tasks: List<String>,
    val dates: List<String>, val rawText: String
)

Математика — Обработка аудио

Параметры аудио потока
fs = 16 000 Гц — частота дискретизации
bits = 16 — разрядность (int16, range: -32768 .. +32767)
channels = 1 (mono)

R_raw = fs × bits/8 × channels = 16000 × 2 × 1 = 32 000 байт/с = 256 кбит/с

BT SPP пропускная способность: ~300–700 кбит/с → достаточно с запасом ≥1.5×
Задержка (latency) сквозной цепочки
t_dma = DMA_BUF_LEN / fs = 512 / 16000 = 32 мс (один DMA буфер)
t_bt ≈ 20–40 мс (BT Classic латентность)
t_parse ≈ 1–2 мс (парсинг пакета на CPU)
t_vosk ≈ 50–150 мс (Vosk partial result на Snapdragon 855+)

t_total = t_dma + t_bt + t_parse + t_vosk ≈ 103–224 мс (субсекундная, неощутима)
VAD порог — расчёт
Типичная речь: RMS ≈ 1000–8000 (16-bit PCM)
Тишина/фон: RMS ≈ 100–400
Порог: E_thresh = 500 — золотая середина

EMA α = 0.3 даёт TC = 1/ln(1/0.7) × 32мс ≈ 89 мс временная константа

Математика — TF-IDF и TextRank

TF-IDF
TF(t, d) = count(t в документе d) / |d|
IDF(t, D) = log( |D| / (1 + |{d ∈ D : t ∈ d}|) ) + 1
TF-IDF(t, d, D) = TF(t,d) × IDF(t,D)

При одном документе: IDF = 1 для всех слов → используем только TF как baseline
При накоплении базы: вычисляем IDF по всему корпусу в SQLite агрегацией
TextRank (адаптированный PageRank)
Схожесть предложений: sim(sᵢ, sⱼ) = |words(sᵢ) ∩ words(sⱼ)| / (log|sᵢ| + log|sⱼ|)

Матрица W нормализована по строкам: Wᵢⱼ = sim(sᵢ, sⱼ) / Σₖ sim(sᵢ, sₖ)

Итерация PageRank:
WS(sᵢ) = (1-d)/N + d × Σⱼ∈In(sᵢ) [ Wⱼᵢ × WS(sⱼ) ]

d = 0.85 (коэф. демпфирования), N = число предложений
Сходимость: 15–25 итераций, ε < 0.0001

Инструкция: pip install mlc-llm, конвертируй модель, добавь .so в jniLibs. Это заменит rule-based NLP на настоящее понимание контекста.
